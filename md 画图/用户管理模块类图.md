    class User {
        +id: number <<PK>>
        +username: string
        +password: string <<encrypted>>
        +avatar: string
        +userType: Enum~"USER|ANCHOR|ADMIN"~
        +createdAt: Date
        +updatedAt: Date
        +getProfile(): UserProfileDTO
        +updateProfile(profileData: UserProfileDTO): Promise~User~
    }

    class UserRepository {
        +findById(id: number): Promise~User|null~
        +findByUsername(username: string): Promise~User|null~
        +save(user: User): Promise~User~
        +update(user: User): Promise~User~
    }

    class AuthUtil {
        +generateToken(userId: number, userType: string): string
        +verifyToken(token: string): valid: boolean, userId?: number 
        +encryptPassword(password: string): string
        +verifyPassword(plain: string, hashed: string): boolean
    }

    class UserService {
        -userRepo: UserRepository
        -authUtil: AuthUtil
        +register(username: string, password: string, userType: string): Promise~User~
        +login(username: string, password: string): Promise user: User, token: string 
        +getUserProfile(userId: number): Promise~UserProfileDTO~
        +updateUserProfile(userId: number, profileData: UserProfileDTO): Promise~User~
        +checkPermission(userId: number, requiredType: string): Promise~boolean~
    }

    class UserAPIClient {
        +baseUrl: string
        +register(formData: RegisterForm): Promise~User~
        +login(credentials: LoginCredentials): Promise user: User, token: string 
        +getProfile(token: string): Promise~UserProfileDTO~
        +updateProfile(token: string, profileData: UserProfileDTO): Promise~User~
    }

    class LoginComponent {
        +formState:  username: string, password: string 
        +errorMessage: string
        +apiClient: UserAPIClient
        +handleInputChange(field: string, value: string): void

    }
